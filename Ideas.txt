# src/mlops/experiment_tracker.py
#
# ПОЧЕМУ: MLflow Tracking — единая точка хранения всех экспериментов.
# Агент НИКОГДА не запускает обучение без активного MLflow run.
# КАК: обёртка-контекст-менеджер гарантирует логирование даже при падении.

import mlflow
import mlflow.sklearn
import mlflow.statsmodels
from contextlib import contextmanager
from pathlib import Path
import yaml


class ExperimentTracker:
    """
    Обёртка над MLflow для унифицированного логирования.
    Агент использует только этот класс, не вызывает mlflow напрямую.
    """

    def __init__(self, config_path: str = "config/mlflow_config.yaml"):
        with open(config_path) as f:
            cfg = yaml.safe_load(f)

        # ПОЧЕМУ: tracking URI вынесен в конфиг, не захардкожен
        mlflow.set_tracking_uri(cfg["tracking_uri"])
        self.experiment_name = cfg["experiment_name"]
        mlflow.set_experiment(self.experiment_name)

    @contextmanager
    def start_run(self, run_name: str, tags: dict = None):
        """
        ПОЧЕМУ: контекст-менеджер гарантирует корректное завершение run
        даже при исключении. КАК: yield внутри with mlflow.start_run()
        """
        with mlflow.start_run(run_name=run_name, tags=tags or {}) as run:
            yield run

    def log_all(
        self,
        params: dict,
        metrics: dict,
        model,
        model_type: str,
        artifacts_dir: Path = None,
        feature_importance: dict = None,
    ):
        """
        ПОЧЕМУ: единый метод логирования — агент не забудет залогировать
        ни один артефакт. КАК: последовательно вызывает все log_* методы.
        """
        # Параметры модели и пайплайна
        mlflow.log_params(params)

        # Метрики качества
        mlflow.log_metrics(metrics)

        # Сохранение модели с сигнатурой входных данных
        if model_type == "sklearn":
            mlflow.sklearn.log_model(model, artifact_path="model")
        elif model_type == "statsmodels":
            mlflow.statsmodels.log_model(model, artifact_path="model")
        elif model_type == "catboost":
            mlflow.catboost.log_model(model, artifact_path="model")

        # Feature importance как отдельный артефакт
        if feature_importance:
            mlflow.log_dict(feature_importance, "feature_importance.json")

        # Дополнительные артефакты (графики, отчёты)
        if artifacts_dir and artifacts_dir.exists():
            mlflow.log_artifacts(str(artifacts_dir))
3.2 Model Registry: стадии жизни модели

# src/mlops/model_registry.py
#
# ПОЧЕМУ: Model Registry — единый источник правды о том,
# какая модель сейчас в проде. Агент не деплоит напрямую из кода обучения.
# КАК: модель проходит стадии Staging → (тесты) → Production → Archived

import mlflow
from mlflow.tracking import MlflowClient


class ModelRegistry:
    """
    Управление жизненным циклом модели через MLflow Model Registry.

    Стадии:
        None       → модель только что залогирована
        Staging    → прошла автотесты, ждёт валидации
        Production → активная боевая модель
        Archived   → выведена из эксплуатации
    """

    # Пороги для автоматического продвижения в Production
    QUALITY_GATES = {
        "mae": 0.42,           # жёсткое требование заказчика
        "mae_improvement": 0.01,  # новая модель должна быть лучше на 1%
    }

    def __init__(self):
        self.client = MlflowClient()
        self.model_name = "liquidity_balance_forecaster"

    def register_model(self, run_id: str) -> str:
        """
        ПОЧЕМУ: регистрируем модель сразу после обучения для версионирования.
        КАК: получаем URI артефакта из run и регистрируем в Registry.
        """
        model_uri = f"runs:/{run_id}/model"
        mv = mlflow.register_model(model_uri, self.model_name)
        return mv.version

    def promote_to_staging(self, version: str):
        """Переводим в Staging — модель готова к тестированию."""
        self.client.transition_model_version_stage(
            name=self.model_name,
            version=version,
            stage="Staging",
            archive_existing_versions=False,
        )

    def promote_to_production(self, version: str, metrics: dict):
        """
        ПОЧЕМУ: автоматическое продвижение только при прохождении
        quality gates. КАК: проверяем MAE и относительное улучшение.
        """
        # Quality Gate 1: абсолютный порог MAE
        if metrics["mae"] > self.QUALITY_GATES["mae"]:
            raise ValueError(
                f"Quality Gate FAILED: MAE={metrics['mae']:.4f} > "
                f"{self.QUALITY_GATES['mae']}. "
                f"Модель НЕ продвинута в Production."
            )

        # Quality Gate 2: улучшение относительно текущей prod-модели
        current_prod = self._get_production_metrics()
        if current_prod:
            improvement = (current_prod["mae"] - metrics["mae"]) / current_prod["mae"]
            if improvement < self.QUALITY_GATES["mae_improvement"]:
                raise ValueError(
                    f"Quality Gate FAILED: улучшение {improvement:.2%} < "
                    f"порога {self.QUALITY_GATES['mae_improvement']:.2%}."
                )

        # Переводим текущую prod в Archived
        self.client.transition_model_version_stage(
            name=self.model_name,
            version=version,
            stage="Production",
            archive_existing_versions=True,   # старая → Archived автоматически
        )

    def _get_production_metrics(self) -> dict | None:
        """Получить метрики текущей Production модели."""
        versions = self.client.get_latest_versions(
            self.model_name, stages=["Production"]
        )
        if not versions:
            return None
        run = self.client.get_run(versions[0].run_id)
        return run.data.metrics

# src/mlops/auto_retrain.py
#
# ПОЧЕМУ: модель деградирует со временем из-за изменения поведения ликвидности.
# Дообучение должно быть полностью автономным — без ручного вмешательства.
# КАК: три независимых триггера → очередь → pipeline дообучения.

from enum import Enum
from datetime import datetime
import yaml


class RetriggerReason(Enum):
    SCHEDULE = "scheduled"           # плановое (раз в N дней)
    DRIFT_DETECTED = "drift"         # детектор разладки сработал
    MAE_DEGRADATION = "mae_drop"     # MAE вырос выше порога в проде


class AutoRetrainOrchestrator:
    """
    Оркестратор автоматического дообучения.
    Определяет: нужно ли дообучать, когда, на каких данных.
    """

    def __init__(self, config_path: str = "config/model_config.yaml"):
        with open(config_path) as f:
            self.cfg = yaml.safe_load(f)["retraining"]

    def check_triggers(self, drift_signal: bool, current_mae: float) -> RetriggerReason | None:
        """
        ПОЧЕМУ: проверяем все триггеры в порядке приоритета.
        Drift > MAE degradation > Schedule.
        КАК: возвращаем причину или None (дообучение не нужно).
        """
        # Триггер 1 (высший приоритет): разладка ряда
        if drift_signal:
            return RetriggerReason.DRIFT_DETECTED

        # Триггер 2: деградация метрики в проде
        if current_mae > self.cfg["mae_alert_threshold"]:
            return RetriggerReason.MAE_DEGRADATION

        # Триггер 3: плановое расписание
        if self._is_scheduled():
            return RetriggerReason.SCHEDULE

        return None  # дообучение не нужно

    def get_training_window(self, reason: RetriggerReason) -> tuple[str, str]:
        """
        ПОЧЕМУ: размер окна дообучения зависит от причины триггера.
        При разладке берём только свежие данные (после точки разладки).
        При плановом — скользящее окно фиксированного размера.
        КАК: возвращаем (date_from, date_to) для фильтрации данных.
        """
        windows = {
            RetriggerReason.SCHEDULE:      self.cfg["window_days_scheduled"],
            RetriggerReason.MAE_DEGRADATION: self.cfg["window_days_degradation"],
            RetriggerReason.DRIFT_DETECTED:  self.cfg["window_days_post_drift"],
        }
        days = windows[reason]
        date_to = datetime.today().strftime("%Y-%m-%d")
        date_from = (datetime.today() - timedelta(days=days)).strftime("%Y-%m-%d")
        return date_from, date_to

# src/serving/api.py
#
# ПОЧЕМУ: каждый прогноз должен логировать метрики для мониторинга.
# Prometheus собирает, Grafana визуализирует, AlertManager уведомляет.
# КАК: FastAPI + prometheus_fastapi_instrumentator + кастомные метрики.

from fastapi import FastAPI
from prometheus_fastapi_instrumentator import Instrumentator
from prometheus_client import Histogram, Counter, Gauge
import mlflow
import numpy as np

app = FastAPI(title="Liquidity Forecast API")

# ─── Кастомные метрики Prometheus ──────────────────────────────
PREDICTION_MAE = Gauge(
    "model_prediction_mae",
    "Скользящий MAE последних N прогнозов"
)
PREDICTION_VALUE = Histogram(
    "model_prediction_value",
    "Распределение предсказанных значений сальдо",
    buckets=[-2, -1, -0.5, 0, 0.5, 1, 2]
)
DRIFT_ALERT = Counter(
    "model_drift_alerts_total",
    "Количество срабатываний детектора разладки"
)
RETRAIN_COUNTER = Counter(
    "model_retrain_total",
    "Количество дообучений",
    ["reason"]   # reason: scheduled | drift | mae_drop
)

# Подключаем стандартные метрики FastAPI (latency, RPS, errors)
Instrumentator().instrument(app).expose(app)


@app.post("/predict")
async def predict(features: dict):
    """
    ПОЧЕМУ: единая точка входа для всех прогнозов.
    КАК: загружаем актуальную prod-модель из MLflow Registry,
    логируем входные данные и предсказание для мониторинга.
    """
    # Загрузка актуальной модели
    model = mlflow.pyfunc.load_model(
        model_uri=f"models:/liquidity_balance_forecaster/Production"
    )
    prediction = model.predict(features)

    # Логируем в Prometheus
    PREDICTION_VALUE.observe(float(prediction))

    return {"prediction": float(prediction), "model_version": model.metadata.run_id}


@app.get("/health")
async def health():
    """Проверка доступности сервиса и актуальности модели."""
    return {"status": "ok", "model": "Production"}

# monitoring/alerts/alert_rules.yml

      # Алерт 2: Детектор разладки сработал
      - alert: DataDriftDetected
        expr: increase(model_drift_alerts_total[1h]) > 0
        labels:
          severity: warning
        annotations:
          summary: "Обнаружена разладка ряда"
          description: "Рассмотреть переключение на ручное управление."

      # Алерт 3: Модель не переобучалась слишком долго
      - alert: ModelStale
        expr: time() - model_last_retrain_timestamp > 86400 * 14
        labels:
          severity: warning

# AGENTS.md — Liquidity Balance Forecasting Agent

> Системная инструкция для AI-агента, реализующего модель прогнозирования
> сальдо ликвидности (разница между притоками и оттоками денежных средств).
> Агент должен строго следовать этому документу на всех этапах разработки.

---

## 1. КОНТЕКСТ ЗАДАЧИ

### Что прогнозируем
- **Целевая переменная**: сальдо ликвидности = притоки − оттоки за день
- **Природа данных**: временные ряды — срезы в обозначенные моменты времени
- **Горизонт**: однодневный прогноз (день вперёд)
- **Целевая точность**: MAE ≤ 0.42 в абсолютном значении (требование заказчика)

### Бизнес-контекст (КРИТИЧНО для понимания цели)
Прогнозная величина используется позиционером (управляющим ликвидностью) для принятия решений:

| Ситуация | Действие | Ставка |
|----------|----------|--------|
| Профицит ликвидности на конец дня | Размещение в ЦБ по overnight | Ключ − 0.9% |
| Дефицит ликвидности на конец дня | Займ по overnight | Ключ + 1% |

**Следствие**: ошибка прогноза имеет асимметричную стоимость.
Агент обязан учитывать это при выборе метрики оптимизации.

---

## 2. ТРЕБОВАНИЯ К МОДЕЛИ

### 2.1 Метрика оптимизации
- [ ] Выбор метрики должен быть **обоснован бизнес-потребностью**
- [ ] Обязательно рассмотреть асимметричные метрики (например, взвешенный MAE)
- [ ] Жёсткое требование заказчика: итоговый MAE ≤ 0.42
- [ ] Каждый блок кода должен содержать комментарий:
  - чем руководствовались при реализации
  - как именно работает блок

### 2.2 Признаки (Feature Engineering)
Агент обязан исследовать следующие группы факторов:

```python
FEATURE_GROUPS = {
    "autoregressive": [
        "лаги таргета (t-1, t-2, ..., t-N)",
        "скользящие средние по таргету",
        "экспоненциальные скользящие средние",
        "rolling std, min, max"
    ],
    "calendar": [
        "день недели",
        "день месяца",
        "даты налоговых дней",        # ВАЖНЫЙ фактор по подсказке заказчика
        "признак конца/начала квартала",
        "признак праздника"
    ],
    "macroeconomic": [
        "ключевая ставка ЦБ",
        "overnight ставки (RUONIA и др.)",
        "прочие макро-факторы"         # исследовать доступность данных
    ]
}
2.3 Отбор признаков (Feature Selection)
 Модель обязана иметь модуль отбора признаков
 Метод отбора должен быть более стабильным относительно альтернатив
Сравнение с baseline-методами обязательно (например, корреляция vs. SHAP vs. Boruta)
Предпочтение: методы, устойчивые к мультиколлинеарности и выбросам
 Результаты отбора логировать и сохранять
2.4 Автоподбор гиперпараметров
 Реализовать автоматический подбор гиперпараметров
 Оптимизация по целевой метрике (не по умолчанию библиотеки)
 Рекомендуемые инструменты: Optuna, Hyperopt, BayesSearchCV
 Результаты каждого запуска сохранять в logs/hyperparam_search/
2.5 Авторегрессионные модели (ОБЯЗАТЕЛЬНО)
⚠️ АГЕНТ ОБЯЗАН исследовать авторегрессионные модели.
Это явное требование заказчика, пропустить нельзя.
 ARIMA / SARIMA
 ARIMAX (с внешними регрессорами)
 VAR (если есть несколько связанных рядов)
 Сравнить с ML-моделями по итоговой метрике
2.6 Калибровка модели
 Подобрать оптимальную частоту переобучения/калибровки
 Если калибровка занимает значительное время — проверить её достаточность (нет ли деградации качества при редкой калибровке)
 Задокументировать выбранную частоту и обоснование
2.7 Автоматическое дообучение
Модель должна дообучаться автоматически.
Все модули работают без ручных корректировок.
 Реализовать pipeline дообучения на новых данных
 Выбор периода дообучения должен быть автоматическим (не захардкоженным)
Рекомендация: на основе детектора разладки (см. п. 2.8) или скользящего окна
 Pipeline логирует каждое дообучение: дата, объём данных, метрика до/после
2.8 Детектор разладки (Drift Detection)
⚠️ КРИТИЧНЫЙ МОДУЛЬ — сигнализирует о необходимости ручного управления.
 Реализовать модуль выявления статистической разладки ряда
 При обнаружении разладки — выдавать сигнал (алерт/флаг)
 Рекомендуемые методы:
CUSUM
Page-Hinkley test
ADWIN
Сравнение распределений (KS-test, PSI)
 Порог чувствительности — настраиваемый параметр

4. ПРАВИЛА ПОВЕДЕНИЯ АГЕНТА
Обязательный порядок работы
1. EDA → 2. Feature Engineering → 3. Feature Selection →
4. Baseline модели (в т.ч. авторегрессионные) →
5. Подбор гиперпараметров → 6. Сравнение моделей →
7. Калибровка/дообучение → 8. Детектор разладки →
9. Финальная валидация (MAE ≤ 0.42?)
Правила написания кода
Каждый блок должен иметь комментарий: # Почему так + # Как работает
Никаких захардкоженных констант — всё в config/model_config.yaml
Все промежуточные результаты логировать
Никаких ручных корректировок в production pipeline
Правила сравнения моделей
Сравнение всегда по одному набору данных (hold-out или time-series CV)
Baseline = наивный прогноз (последнее известное значение)
Финальная таблица сравнения обязательна перед выбором модели
Стоп-сигналы (агент останавливается и сообщает)
MAE > 0.42 на тестовой выборке — не деплоить, сообщить и предложить улучшения
Детектор разладки сработал — выдать алерт, не переобучать автоматически
Данные содержат пропуски > 20% — остановиться, запросить решение
5. КРИТЕРИИ ГОТОВНОСТИ (Definition of Done)
 MAE ≤ 0.42 на тестовой выборке
 Авторегрессионные модели исследованы и задокументированы
 Модуль отбора признаков реализован и сравнён с альтернативами
 Гиперпараметры подобраны автоматически
 Pipeline дообучения работает без ручного вмешательства
 Детектор разладки протестирован на синтетических данных
 Все блоки кода прокомментированы (почему + как)
 Частота калибровки обоснована и задокументирована
 Все результаты воспроизводимы (зафиксированы random seeds)
